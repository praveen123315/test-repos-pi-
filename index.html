Google Sheets Integration

Connect this app to Google Sheets. Once the admin sets the URL below and re-shares the HTML file, all team members automatically connect — no setup needed on their side.

1
Create a Google Sheet with two tabs: PI Data and Quotation Data.
Row 1 headers: Doc No | Type | Date | Buyer | GSTIN | State | Address | Payment Terms | Items | Subtotal | Tax | Total | GST Type | Freight | Remark | Status | PDF Link | Saved At
2
Extensions → Apps Script → replace ALL code with this (handles both data saving AND PDF upload to Google Drive):
var SHEET_PI = "PI Data"; var SHEET_QT = "Quotation Data"; var FOLDER_NAME = "Surface Paints PDFs"; function doPost(e) { try { var p = e.parameter; var action = p.action || ""; var type = p.type || "PI"; var sheetName = type === "PI" ? SHEET_PI : SHEET_QT; var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName); var pdfLink = ""; // Upload PDF to Drive if included if (p.pdf_data && p.pdf_filename) { var folder; var it = DriveApp.getFoldersByName(FOLDER_NAME); folder = it.hasNext() ? it.next() : DriveApp.createFolder(FOLDER_NAME); var bytes = Utilities.base64Decode(p.pdf_data); var blob = Utilities.newBlob(bytes, "application/pdf", p.pdf_filename); var file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); pdfLink = file.getUrl(); } // Save row (action = "save" or "savewithpdf") if (action === "save" || action === "savewithpdf") { sheet.appendRow([ p.doc_no, type, p.date, p.buyer_name, p.buyer_gstin, p.buyer_state, p.buyer_address, p.payment_terms, p.items, parseFloat(p.sub)||0, parseFloat(p.tax)||0, parseFloat(p.total)||0, p.gst_type, p.freight, p.remark, "Saved", pdfLink, p.saved_at ]); } // Update existing row by Doc No (action = "update" or "updatewithpdf") if (action === "update" || action === "updatewithpdf") { var data = sheet.getDataRange().getValues(); var headers = data[0]; var docNoCol = headers.indexOf("Doc No"); var rowFound = -1; for (var i = 1; i < data.length; i++) { if (String(data[i][docNoCol]) === String(p.doc_no)) { rowFound = i + 1; break; } } var newRow = [ p.doc_no, type, p.date, p.buyer_name, p.buyer_gstin, p.buyer_state, p.buyer_address, p.payment_terms, p.items, parseFloat(p.sub)||0, parseFloat(p.tax)||0, parseFloat(p.total)||0, p.gst_type, p.freight, p.remark, "Updated", pdfLink || (rowFound > 0 ? data[rowFound-1][headers.indexOf("PDF Link")] : ""), p.saved_at ]; if (rowFound > 0) { sheet.getRange(rowFound, 1, 1, newRow.length).setValues([newRow]); } else { sheet.appendRow(newRow); // fallback: append if not found } } return ContentService .createTextOutput(JSON.stringify({ success: true, pdf_link: pdfLink })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { // History tab: return all records var filterType = (e.parameter && e.parameter.type) ? e.parameter.type : "ALL"; var ss = SpreadsheetApp.getActiveSpreadsheet(); var records = []; var names = filterType==="PI" ? [SHEET_PI] : filterType==="QT" ? [SHEET_QT] : [SHEET_PI,SHEET_QT]; names.forEach(function(sn){ var sh = ss.getSheetByName(sn); if(!sh) return; var rows = sh.getDataRange().getValues(); var headers = rows[0]; rows.slice(1).forEach(function(r){ var obj = {}; headers.forEach(function(h,i){ obj[h]=r[i]; }); records.push(obj); }); }); return ContentService .createTextOutput(JSON.stringify(records)) .setMimeType(ContentService.MimeType.JSON); }
